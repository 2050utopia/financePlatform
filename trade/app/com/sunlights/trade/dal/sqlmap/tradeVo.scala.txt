@(name:String, qmap:Map[String,Object])
@name match {
	case "TradeDao.getTradeListByCustomerId" =>{
	    @if(qmap.get("productType") == "0" || qmap.get("productType") == "1") {
            select  new com.sunlights.trade.vo.TradeInfoVo(
                        f.fundCode,pt.chiName,hp.subCapital,hp.yesterdayProfit,
                        to_char(t.tradeTime, 'YYYY-MM-DD HH24:MI'),
                        t.type,t.tradeStatus,t.payStatus,t.confirmStatus,
                        t.tradeNo,
                        t.bankName,
                        t.bankCardNo
                    )
            from    Trade t,HoldCapital hp,Product pt,Fund f
            where   t.holdCapitalId = hp.id
            and     t.custId = :customerId
            and     t.prdCode = f.fundCode
            and     f.fundCode = pt.innerCode
            order by t.tradeTime desc
	    }
	}


}
